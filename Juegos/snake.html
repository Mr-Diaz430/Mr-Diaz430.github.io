<Snake (C++)/>

Bienvenido al apartado de juegos, En esta sección te mostraremos la codificación de un juego en particular, con la intención de que seas capaz de presenciar un poco de lo que es posible de realizar con un lenguaje de programación.

¿Qué es el Snake?

El Snake es un videojuego lanzado a mediados de la década de 1970 que ha mantenido su popularidad desde entonces convirtiéndose en un clásico. Lo animamos a probar el juego desde tu entorno de desarrollo, lo único que debe hacer es copiar el código del procedimiento y si es de su agrado experimentar y tratar de comprender su funcionamiento.

Como jugarlo

El juego de la serpiente es un juego sencillo en el que una serpiente se mueve adentro de una caja y trata de comerse una manzana. Una vez que se come la manzana, la serpiente crece de largo y se mueve más rápido, el juego termina una vez que la serpiente se choca consigo misma o con cualquiera de las cuatro paredes.

Controles: flechas del teclado.

Código

Primero importamos las librerías y definimos las teclas que usaremos


      #include <windows.h>
      #include <iostream>
      #include <stdlib.h>
      #include <conio.h>
          
      #define ARRIBA 72
      #define IZQUIERDA 75
      #define DERECHA 77
      #define ABAJO 80
      #define ESC 27

Declaración de variables


      int cuerpo[200][2];
      int n = 1, tam = 10, dir = 3; //Guardar partes del cuerpo, Tamaño del cuerpo, y la direccion
      int x = 10, y = 12;
      int xc = 30, yc = 15; //Comida variables
      int velocidad = 60;
      char tecla;

Usaremos las función VOID GOTOXY para mover el cursor de la ventana de ptexto a la posición según las coordenadas especificadas por los argumentos x e y.


      void gotoxy(int x, int y) { 
        HANDLE hCon; 
        COORD dwPos; 
          
        dwPos.X = x; 
        dwPos.Y = y; 
        hCon = GetStdHandle(STD_OUTPUT_HANDLE); 
        SetConsoleCursorPosition(hCon,dwPos); 
      }

Función para quitar el puntero del mouse.


      void OcultaCursor() {
        CONSOLE_CURSOR_INFO cci = {100, FALSE};
              
        SetConsoleCursorInfo(GetStdHandle(STD_OUTPUT_HANDLE), &cci);
      }

Usaremos void para crear un marco para el juego


      void pintar() {  //Lineas horizontales
        for(int i=2; i < 78; i++){
        gotoxy (i, 3); printf ("%c", 205);
        gotoxy(i, 23); printf ("%c", 205); 
      }
      
      for(int v=4; v < 23; v++){  //Lineas verticales
        gotoxy (2,v);  printf ("%c", 186);
        gotoxy(77,v);  printf ("%c", 186); 
      }

      gotoxy  (2,3);    printf ("%c", 201); //Esquinas
      gotoxy (2,23);    printf ("%c", 200);
      gotoxy (77,3);    printf ("%c", 187);
      gotoxy(77,23);    printf ("%c", 188); 
      }

Usaremos matriz para guardar la posición del cuerpo de la serpiente


      void guardar_posicion(){
        cuerpo[n][0] = x;
        cuerpo[n][1] = y;

        n++;
      
         if(n == tam) n = 1;
      }

Crearemos el cuerpo de la serpiente con void para que la serpiente crezca.


      void dibujar_cuerpo(){
        for(int i = 1; i < tam; i++){
        gotoxy(cuerpo[i][0] , cuerpo[i][1]); printf("*");
         }
       }

Función para borrar el cuerpo.


      void borrar_cuerpo(){
        gotoxy(cuerpo[n][0] , cuerpo[n][1]); printf(" ");
      }

Función que sirve para poder mover las teclas.


      void teclear(){
        
        if(kbhit()){
          tecla = getch();
          switch(tecla){
            case ARRIBA : if(dir != 2) dir = 1; break;
            case ABAJO : if(dir != 1) dir = 2; break;
            case DERECHA : if(dir != 4) dir = 3; break;
            case IZQUIERDA : if(dir != 3) dir = 4; break;
          }
        }
      }

Función Comida.


      void comida() {
        if(x == xc && y == yc) {
                
          xc = (rand() % 73) + 4;
          yc = (rand() % 19) + 4;
                
          tam++;
          gotoxy(xc, yc); printf("%c", 4);
        }
      }

Función para cuando perdamos.


      bool game_over() {
                 
        if(y == 3 || y == 23 || x == 2 || x == 77) return false;

          for(int j = tam - 1; j > 0; j--){
            if(cuerpo[j][0] == x &&  cuerpo[j][1] == y)
            return false;
          }
          return true;

        }

Programa Principal


        int main() {
          
          OcultaCursor();
          pintar();
          gotoxy(xc, yc); printf("%c", 4);
              
          while(tecla != ESC && game_over()) {
             
            borrar_cuerpo();
            guardar_posicion();
            dibujar_cuerpo();
            comida();
            teclear();
            teclear();
                
            if(dir == 1) y--;
            if(dir == 2) y++;
            if(dir == 3) x++;
            if(dir == 4) x--;
                
            Sleep(velocidad);
          }

          pintar();
          return 0;
        }
